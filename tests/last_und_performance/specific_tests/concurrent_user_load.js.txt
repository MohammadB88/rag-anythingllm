// concurrent_user_load.js
// Simulates 90 concurrent users/min with realistic think time
// Ramps up gradually to find the degradation point

import http from "k6/http";
import { check, sleep } from "k6";
import { Trend, Rate, Gauge } from "k6/metrics";
import { SharedArray } from "k6/data";

const TTFT = new Trend("ttft_ms", true);
const errorRate = new Rate("error_rate");
const activeUsers = new Gauge("active_users");

const BASE_URL = __ENV.BASE_URL || "https://your-model-endpoint";
const API_KEY = __ENV.API_KEY || "your-api-key";
const MODEL = __ENV.MODEL || "gpt-oss-120b";

// Loaded once at init time and shared across all VUs
const PROMPTS = new SharedArray("prompts", function () {
  return JSON.parse(open("./prompts.json"));
});

export const options = {
  scenarios: {
    // Stage 1: Ramp from 0 → baseline (64 VUs ~ avg load)
    ramp_to_baseline: {
      executor: "ramping-vus",
      startVUs: 0,
      stages: [
        { duration: "3m", target: 64 },   // ramp to avg
        { duration: "5m", target: 64 },   // hold at avg
        { duration: "2m", target: 90 },   // ramp to observed peak users
        { duration: "5m", target: 90 },   // hold at peak
        { duration: "2m", target: 120 },  // push 33% above peak (stress)
        { duration: "3m", target: 120 },  // hold overshoot
        { duration: "3m", target: 0 },    // ramp down
      ],
    },
  },
  thresholds: {
    ttft_ms: [
      "p(95)<5000",   // peak SLO
      "p(50)<2000",   // median target
    ],
    http_req_duration: ["p(99)<45000"],
    error_rate: ["rate<0.01"],
  },
};

export default function () {
  activeUsers.add(1);

  const prompt = PROMPTS[Math.floor(Math.random() * PROMPTS.length)];

  const payload = JSON.stringify({
    model: MODEL,
    messages: [
      { role: "system", content: prompt.system },
      { role: "user", content: prompt.user },
    ],
    max_tokens: 2800,
    stream: false,
  });

  const params = {
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${API_KEY}`,
    },
    timeout: "90s",  // Higher timeout to capture slow responses, not just drop them
  };

  const start = Date.now();
  const res = http.post(`${BASE_URL}/v1/chat/completions`, payload, params);
  const elapsed = Date.now() - start;

  const ok = check(res, {
    "status 200": (r) => r.status === 200,
    "no timeout": (r) => r.timings.duration < 60000,
    "valid response": (r) => {
      try { return JSON.parse(r.body).choices?.length > 0; } catch { return false; }
    },
  });

  errorRate.add(!ok);
  TTFT.add(elapsed);

  activeUsers.add(-1);

  // Realistic think time — users read responses before sending next request
  // Drawn from uniform distribution: 8–20s (matches <1 req/min per user pattern)
  sleep(Math.random() * 12 + 8);
}
