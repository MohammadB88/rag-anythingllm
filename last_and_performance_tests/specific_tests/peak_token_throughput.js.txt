// peak_token_throughput.js
// Stresses the observed peak: 2.33M input tokens/min + 210K output tokens/min
// Primary target: expose prefill bottleneck on the 120B model
// NOTE: You need ~74 concurrent requests/min of ~31.5K tokens each to hit peak

import http from "k6/http";
import { check, sleep } from "k6";
import { Trend, Rate, Counter } from "k6/metrics";
import { SharedArray } from "k6/data";

const TTFT = new Trend("ttft_ms", true);
const prefillThroughput = new Trend("input_tokens_per_min");
const decodeThroughput = new Trend("output_tokens_per_sec");
const errorRate = new Rate("error_rate");
const inputTokenCounter = new Counter("total_input_tokens");
const outputTokenCounter = new Counter("total_output_tokens");

const BASE_URL = __ENV.BASE_URL || "https://your-model-endpoint";
const API_KEY = __ENV.API_KEY || "your-api-key";
const MODEL = __ENV.MODEL || "gpt-oss-120b";

// ⚠️  prompts.json entries should include an estimated_input_tokens field.
//     Target: each prompt should be ~31,500 tokens (input) to match peak stats.
// Loaded once at init time and shared across all VUs
const LARGE_PROMPTS = new SharedArray("prompts", function () {
  return JSON.parse(open("./prompts.json"));
});

export const options = {
  scenarios: {
    // Phase 1: Ramp to peak token throughput
    token_peak_ramp: {
      executor: "ramping-arrival-rate",
      startRate: 10,
      timeUnit: "1m",
      stages: [
        { duration: "3m", target: 40 },   // 50% of peak req rate
        { duration: "3m", target: 64 },   // avg req rate
        { duration: "3m", target: 74 },   // peak req rate (matches 2.33M tok/min)
        { duration: "5m", target: 74 },   // sustain peak
        { duration: "3m", target: 100 },  // overshoot — find breaking point
        { duration: "3m", target: 74 },   // recover back to peak
        { duration: "2m", target: 0 },    // ramp down
      ],
      preAllocatedVUs: 80,
      maxVUs: 150,
    },
  },
  thresholds: {
    ttft_ms: [
      "p(50)<2000",
      "p(95)<5000",
      "p(99)<10000",  // Even tail latency tracked
    ],
    http_req_duration: ["p(95)<30000"],
    error_rate: ["rate<0.01"],
    // Custom: ensure we're actually hitting token targets
    // (validate post-run: total_input_tokens / test_duration_min should ≈ 2.33M)
  },
};

export default function () {
  const prompt = LARGE_PROMPTS[Math.floor(Math.random() * LARGE_PROMPTS.length)];

  const payload = JSON.stringify({
    model: MODEL,
    messages: [
      { role: "system", content: prompt.system },
      { role: "user", content: prompt.user },
    ],
    max_tokens: 2800,
    stream: false,
  });

  const params = {
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${API_KEY}`,
    },
    timeout: "120s",  // Long timeout — peak load may queue requests
  };

  const start = Date.now();
  const res = http.post(`${BASE_URL}/v1/chat/completions`, payload, params);
  const elapsed = Date.now() - start;

  const ok = check(res, {
    "status 200": (r) => r.status === 200,
    "choices present": (r) => {
      try { return JSON.parse(r.body).choices?.length > 0; } catch { return false; }
    },
  });

  errorRate.add(!ok);
  TTFT.add(elapsed);

  if (ok) {
    try {
      const body = JSON.parse(res.body);
      const inputTok = body.usage?.prompt_tokens || prompt.estimated_input_tokens;
      const outputTok = body.usage?.completion_tokens || 0;

      inputTokenCounter.add(inputTok);
      outputTokenCounter.add(outputTok);

      // Track decode speed (output tokens per second)
      if (elapsed > 0 && outputTok > 0) {
        decodeThroughput.add((outputTok / elapsed) * 1000);
      }
    } catch (_) {}
  }

  // Minimal think time — this test targets throughput, not user behavior
  // Small sleep to prevent VU spin-loops but keep pressure high
  sleep(0.5);
}
